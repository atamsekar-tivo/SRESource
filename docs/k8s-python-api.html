🖥️ Nodes API

List All Nodes

v1.list_node()
List all nodes in the cluster
from kubernetes import client, config

# Load kubeconfig
config.load_kube_config()
v1 = client.CoreV1Api()

# List all nodes
nodes = v1.list_node()
for node in nodes.items:
    print(f"Node: {node.metadata.name}")
    print(f"Status: {node.status.conditions[-1].type}")

Get Node Details

v1.read_node(name)
Get detailed information about a specific node
Parameters:
• name: Node name (required) 
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

# Get specific node
node = v1.read_node(name="node-1")
print(f"Allocatable CPU: {node.status.allocatable['cpu']}")
print(f"Allocatable Memory: {node.status.allocatable['memory']}")

# Check node conditions
for condition in node.status.conditions:
    print(f"{condition.type}: {condition.status}")

Get Node Resources

v1.read_node(name).status.allocatable
Check CPU, memory, and storage allocatable on a node
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

node = v1.read_node(name="worker-1")

# Get allocatable resources
allocatable = node.status.allocatable
print("Allocatable Resources:")
print(f"  CPU: {allocatable['cpu']}")
print(f"  Memory: {allocatable['memory']}")
print(f"  Pods: {allocatable['pods']}")

# Get requested resources
requested = node.status.requested
print("Requested Resources:")
print(f"  CPU: {requested.get('cpu', 'N/A')}")
print(f"  Memory: {requested.get('memory', 'N/A')}")
💡 Example: Monitor Node Health 
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

def check_node_health(node_name):
    node = v1.read_node(name=node_name)
    conditions = {c.type: c.status for c in node.status.conditions}
    
    if conditions.get('Ready') == 'True':
        print(f"✓ {node_name} is Ready")
    else:
        print(f"✗ {node_name} is NotReady")
    
    return conditions

# Check all nodes
nodes = v1.list_node()
for node in nodes.items:
    check_node_health(node.metadata.name)

📦 Pods API

List Pods

v1.list_namespaced_pod(namespace, watch=False)
List all pods in a namespace
Parameters:
• namespace: Namespace name (required)
• watch: Watch for changes (optional, default: False)
• label_selector: Filter by labels (optional) 
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

# List all pods in default namespace
pods = v1.list_namespaced_pod(namespace="default")
for pod in pods.items:
    print(f"Pod: {pod.metadata.name}")
    print(f"Status: {pod.status.phase}")
    print(f"Node: {pod.spec.node_name}")

# List pods with label selector
selector = "app=nginx"
pods = v1.list_namespaced_pod(namespace="default", label_selector=selector)
print(f"Found {len(pods.items)} nginx pods")

Get Pod Details

v1.read_namespaced_pod(name, namespace)
Get detailed information about a specific pod
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

pod = v1.read_namespaced_pod(name="my-pod", namespace="default")

# Pod information
print(f"Name: {pod.metadata.name}")
print(f"Status: {pod.status.phase}")
print(f"IP: {pod.status.pod_ip}")
print(f"Node: {pod.spec.node_name}")

# Container information
for container in pod.spec.containers:
    print(f"Container: {container.name}")
    print(f"Image: {container.image}")

# Container status
for status in pod.status.container_statuses:
    print(f"Container: {status.name} - Ready: {status.ready}")

Get Pod Logs

v1.read_namespaced_pod_log(name, namespace, container=None)
Retrieve logs from a pod
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

# Get pod logs
try:
    logs = v1.read_namespaced_pod_log(
        name="my-pod",
        namespace="default",
        container="my-container",
        tail_lines=100
    )
    print(logs)
except Exception as e:
    print(f"Error: {e}")

# Get previous logs (if pod restarted)
logs = v1.read_namespaced_pod_log(
    name="my-pod",
    namespace="default",
    previous=True
)
💡 Example: Debug Pod Status 
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

def debug_pod(pod_name, namespace):
    pod = v1.read_namespaced_pod(name=pod_name, namespace=namespace)
    
    print(f"Pod Status: {pod.status.phase}")
    
    if pod.status.phase != "Running":
        # Get events
        events = v1.list_namespaced_event(namespace=namespace)
        for event in events.items:
            if event.involved_object.name == pod_name:
                print(f"Event: {event.reason} - {event.message}")
    
    # Check container status
    for status in pod.status.container_statuses or []:
        print(f"Container: {status.name}")
        print(f"  Ready: {status.ready}")
        print(f"  Restarts: {status.restart_count}")
        if status.last_state and status.last_state.terminated:
            print(f"  Exit Code: {status.last_state.terminated.exit_code}")

debug_pod("my-pod", "default")

Scale Deployment

apps_v1.patch_namespaced_deployment_scale(name, namespace, body)
Scale a deployment to desired replicas
from kubernetes import client, config

config.load_kube_config()
apps_v1 = client.AppsV1Api()

# Scale deployment to 5 replicas
body = {
    "spec": {
        "replicas": 5
    }
}

result = apps_v1.patch_namespaced_deployment_scale(
    name="my-app",
    namespace="default",
    body=body
)
print(f"Scaled to {result.spec.replicas} replicas")

from kubernetes import client, config

if svc.status.load_balancer and svc.status.load_balancer.ingress:

# Get ports

Read Secret

v1.read_namespaced_secret(name, namespace)
Get secret data (base64 encoded)
from kubernetes import client, config
import base64

config.load_kube_config()
v1 = client.CoreV1Api()

secret = v1.read_namespaced_secret(name="my-secret", namespace="default")

# Decode secret data
for key, encoded_value in (secret.data or {}).items():
    decoded_value = base64.b64decode(encoded_value).decode('utf-8')
    print(f"{key}: {decoded_value}")

Create Namespace

v1.create_namespace(body)
Create a new namespace
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

body = client.V1Namespace(
    metadata=client.V1ObjectMeta(name="my-namespace")
)

try:
    v1.create_namespace(body)
    print("Namespace created")
except client.rest.ApiException as e:
    print(f"Error: {e}")
